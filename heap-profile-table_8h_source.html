60  // Profile stats.
61  struct Stats {
62  int32 allocs; // Number of allocation calls
63  int32 frees; // Number of free calls
64  int64 alloc_size; // Total size of all allocated objects so far

67  // semantic equality
68  bool Equivalent(const Stats& x) const {
69  return allocs - frees == x.allocs - x.frees &&
70  alloc_size - free_size == x.alloc_size - x.free_size;

193  // Hash table bucket to hold (de)allocation stats
194  // for a given allocation call stack trace.
195  struct Bucket : public Stats {
196  uintptr_t hash; // Hash value of the stack trace
197  int depth; // Depth of stack trace

202  // Info stored in the address map
203  struct AllocValue {
204  // Access to the stack-trace bucket
205  Bucket* bucket() const {
206  return reinterpret_cast<Bucket*>(bucket_rep & ~uintptr_t(kMask));
207  }
208  // This also does set_live(false).
209  void set_bucket(Bucket* b) { bucket_rep = reinterpret_cast<uintptr_t>(b); }
210  size_t bytes; // Number of bytes in this allocation
211 
212  // Access to the allocation liveness flag (for leak checking)
213  bool live() const { return bucket_rep & kLive; }
214  void set_live(bool l) {
215  bucket_rep = (bucket_rep & ~uintptr_t(kLive)) | (l ? kLive : 0);
216  }
217 
218  // Should this allocation be ignored if it looks like a leak?
219  bool ignore() const { return bucket_rep & kIgnore; }
220  void set_ignore(bool r) {
221  bucket_rep = (bucket_rep & ~uintptr_t(kIgnore)) | (r ? kIgnore : 0);
222  }
223 
224  private:
225  // We store a few bits in the bottom bits of bucket_rep.
226  // (Alignment is at least four, so we have at least two bits.)
227  static const int kLive = 1;
228  static const int kIgnore = 2;
229  static const int kMask = kLive | kIgnore;
230 
231  uintptr_t bucket_rep;
232  };

270  // Helper for IterateAllocs to do callback signature conversion
271  // from AllocationMap::Iterate to AllocIterator.
272  static void MapArgsAllocIterator(const void* ptr, AllocValue* v,

274  AllocInfo info;

276  info.call_stack = v->bucket()->stack;

361  // Total count/size are stored in a Bucket so we can reuse UnparseBucket

365  // own object->bucket map.
366  AllocationMap map_;

369  memset(&total_, 0, sizeof(total_));
