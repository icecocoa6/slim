/* bt_idxより上のidxに積まれたデータをログ上からunputしていく */ \
while (vec_num(&((Tracker)->traced_ids)) > __bt_idx) { \
LmnWord __key = vec_pop(&((Tracker)->traced_ids)); \
UnputFunc(Tbl, __key); \
} \
} while (0)
