45 // Generic allocator class for STL objects
46 // that uses a given type-less allocator Alloc, which must provide:
47 // static void* Alloc::Allocate(size_t size);
48 // static void Alloc::Free(void* ptr, size_t size);
49 //
50 // STL_Allocator<T, MyAlloc> provides the same thread-safety
51 // guarantees as MyAlloc.
52 //
53 // Usage example:
54 // set<T, less<T>, STL_Allocator<T, MyAlloc> > my_set;
55 // CAVEAT: Parts of the code below are probably specific
56 // to the STL version(s) we are using.
57 // The code is simply lifted from what std::allocator<> provides.
58 template <typename T, class Alloc>
59 class STL_Allocator {
60  public:
61  typedef size_t size_type;
62  typedef ptrdiff_t difference_type;
63  typedef T* pointer;
64  typedef const T* const_pointer;
65  typedef T& reference;
66  typedef const T& const_reference;
67  typedef T value_type;
68 
69  template <class T1> struct rebind {
70  typedef STL_Allocator<T1, Alloc> other;
71  };
72 
73  STL_Allocator() { }
74  STL_Allocator(const STL_Allocator&) { }
75  template <class T1> STL_Allocator(const STL_Allocator<T1, Alloc>&) { }
76  ~STL_Allocator() { }
77 
78  pointer address(reference x) const { return &x; }
79  const_pointer address(const_reference x) const { return &x; }
80 
81  pointer allocate(size_type n, const void* = 0) {
82  RAW_DCHECK((n * sizeof(T)) / sizeof(T) == n, "n is too big to allocate");
83  return static_cast<T*>(Alloc::Allocate(n * sizeof(T)));
84  }
85  void deallocate(pointer p, size_type n) { Alloc::Free(p, n * sizeof(T)); }
86 
87  size_type max_size() const { return size_t(-1) / sizeof(T); }
88 
89  void construct(pointer p, const T& val) { ::new(p) T(val); }
90  void destroy(pointer p) { p->~T(); }
91 
92  // There's no state, so these allocators are always equal
93  bool operator==(const STL_Allocator&) const { return true; }
94 };
