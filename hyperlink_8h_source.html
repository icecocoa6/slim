65  * '!'をファンクタに持つシンボルアトムごとに生成される
66  * '!'アトムのポインタをIDとして利用しているが、出力（とuniqの履歴生成）の際には
67  * idの値を見かけ上のIDとして利用している
68  */

75  LmnAtomRef attrAtom;/* ハイパーリンクの属性として扱うアトム rootにのみ持たせる */

77  /* 木構造による併合関係の表現 */
78  struct HyperLink *parent; /* root の場合は自身のポインタ */
79  struct HashSet *children; /* 子表 */

81 } HyperLink;
