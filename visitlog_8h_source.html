273 static inline void sproc_tbl_expand(SimplyProcessTableRef p, unsigned long n) {
274  unsigned int org_n = p->num_buckets;
275  while (p->cap <= n) p->cap *= 2;
276  p->num_buckets = p->cap / PROC_TBL_BUCKETS_SIZE + 1;
277 
278  if (org_n < p->num_buckets) {
279  p->tbl = LMN_REALLOC(BYTE *, p->tbl, p->num_buckets);
280  memset(p->tbl + org_n, 0, sizeof(BYTE *) * (p->num_buckets - org_n));
281  }
282 
283  unsigned int b = n / PROC_TBL_BUCKETS_SIZE;
284  if (b < p->num_buckets && p->tbl[b]) return;
285  p->tbl[b] = LMN_NALLOC(BYTE, PROC_TBL_BUCKETS_SIZE);
286  memset(p->tbl[b], SPROC_TBL_INIT_V, sizeof(BYTE) * PROC_TBL_BUCKETS_SIZE);
287 }

424 #define LogTracker_REVERT(Tracker, UnputFunc, Tbl) \
425  do { \
426  unsigned long __bt_idx; \
427  LMN_ASSERT(!vec_is_empty(&((Tracker)->btp_idx))); \
428  \
429  /* 前回セットしたトレース数のメモを取り出す */ \
430  __bt_idx = vec_pop(&((Tracker)->btp_idx)); \
431  /* bt_idxより上のidxに積まれたデータをログ上からunputしていく */ \
432  while (vec_num(&((Tracker)->traced_ids)) > __bt_idx) { \
433  LmnWord __key = vec_pop(&((Tracker)->traced_ids)); \
434  UnputFunc(Tbl, __key); \
435  } \
436  } while (0)

465  ProcessID matched; /* 対応させているプロセスとマッチさせたプロセスのID.

569 /* トレースログlのテーブルサイズをnew_size以上の大きさに拡張する. */
570 static inline void tracelog_tbl_expand(TraceLogRef l, unsigned long new_size)
571 {
572  unsigned int org_n = l->num_buckets;
573 
574  while (l->cap <= new_size) l->cap *= 2;
575  l->num_buckets = l->cap / PROC_TBL_BUCKETS_SIZE + 1;
576  if (org_n < l->num_buckets) {
577  l->tbl = LMN_REALLOC(struct TraceData *, l->tbl, l->num_buckets);
578  memset(l->tbl + org_n, 0, sizeof(struct TraceData *) * (l->num_buckets - org_n));
579  }
580 
581  unsigned int b = new_size / PROC_TBL_BUCKETS_SIZE;
582  if (b < l->num_buckets && l->tbl[b]) return;
583  l->tbl[b] = LMN_NALLOC(struct TraceData, PROC_TBL_BUCKETS_SIZE);
584  memset(l->tbl[b], TLOG_INIT_DATA, sizeof(struct TraceData) * PROC_TBL_BUCKETS_SIZE);
585 }

587 /* ログl上のキーkey位置にあるTraceLogに対して, 訪問を記録する.
588  * matched_idはマッチしたプロセスのID, ownerは所属膜.
589  * 所属膜がNULLでない場合は, 所属膜の情報を記録し, 所属膜側のプロセス訪問カウンタを回す.
590  * tracelog_put_atom, tracelog_put_memから呼び出す関数であり, 直接callしないこと.
591  * (呼出しコスト削減のために公開関数としているだけ) */
592 static inline int tracelog_put(TraceLogRef l, LmnWord key, LmnWord matched_id,
593  LmnMembraneRef owner) {
594  tracelog_tbl_expand(l, key);
595 
596  if (TLOG_IS_TRV(TLOG_FLAG(tracelog_entry(l, key)))) {
597  return 0;
598  }
599 
600  l->num++;
601  TLOG_SET_MATCHED(tracelog_entry(l, key), matched_id);
602 
603  if (owner) {
604  TLOG_SET_OWNER(tracelog_entry(l, key), owner);
605  LMN_ASSERT(l->cap > lmn_mem_id(owner));
606  tracelog_tbl_expand(l, lmn_mem_id(owner));
607  TLOG_TRV_INC(tracelog_entry(l, lmn_mem_id(owner)));
608  }
609 
610  LogTracker_TRACE(&l->tracker, key);
611 
612  return 1;
613 }
