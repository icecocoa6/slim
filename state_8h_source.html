89 struct State { /* Total:72(36)byte */
90  unsigned int successor_num; /* 4(4)byte: サクセッサの数 */
91  BYTE state_name; /* 1(1)byte: 同期積オートマトンの性質ラベル */
92  BYTE flags; /* 1(1)byte: フラグ管理用ビットフィールド */
93  BYTE flags2; /* 1(1)byte: フラグ管理用ビットフィールド2 */
94  BYTE flags3; /* 1(1)byte: アラインメントの隙間(一時的にdpor_naiveで使用中) */
95  unsigned long hash; /* 8(4)byte: 通常時: 膜memのハッシュ値, --mem-enc時: 膜の一意なバイト列のハッシュ値 */
96  state_data_t data; /* 8(4)byte: 膜, バイナリストリングのどちらか */
97  TreeCompressData tcd; /* 8(8)byte: Tree Compression 用のデータ無理やり8 Byteにしている */
98  succ_data_t *successors; /* 8(4)byte: サクセッサポインタの配列 */
99  State *next; /* 8(4)byte: 状態管理表に登録する際に必要なポインタ */
100  State *parent; /* 8(4)byte: 自身を生成した状態へのポインタを持たせておく */
101  unsigned long state_id; /* 8(4)byte: 生成順に割り当てる状態の整数ID */
102  State *map; /* 8(4)byte: MAP値 or 最適化実行時の前状態 */
103 #ifndef MINIMAL_STATE
104  BYTE *local_flags; /* 8(4)byte: 並列実行時、スレッド事に保持しておきたいフラグ(mcndfsのcyanフラグ等) */
105  pthread_mutex_t expand_lock;
106  unsigned long expander_id;
107 #endif
108 #ifdef KWBT_OPT
109  LmnCost cost; /* 8(4)byte: cost */
110 #endif
111 };

411 /* delta-compression用のinner関数.
412  * flagが真の場合, デコード済みのバイナリストリングをキャッシュから取得する.
413  * キャッシュにバイナリストリングを置かないケースで使用する場合は,
414  * inner関数を直接呼び出し, flagに偽を渡しておけばよい. */
415 static inline LmnMembraneRef state_restore_mem_inner(State *s, BOOL flag) {
416  if (state_mem(s)) {
417  return state_mem(s);
418  }
419  else if (s_is_d(s)) {
420  LmnBinStrRef b;
421  if (flag) {
422  b = state_D_fetch(s);
423  } else {
424  b = state_binstr_reconstructor(s);
425  }
426  return lmn_binstr_decode(b);
427  }
428  else {
429  LmnBinStrRef b = state_binstr(s);
430  if (lmn_env.tree_compress && b == NULL) {
431  TreeNodeID ref;
432  tcd_get_root_ref(&s->tcd, &ref);
433  LMN_ASSERT(ref);
434  LMN_ASSERT(tcd_get_byte_length(&s->tcd) != 0);
435  b = lmn_bscomp_tree_decode((TreeNodeID)ref, tcd_get_byte_length(&s->tcd));
436  }
437  LMN_ASSERT(b);
438  return lmn_binstr_decode(b);
439  }
440 }
