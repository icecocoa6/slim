40 #define FUTEX_WAIT 0

44 static bool have_futex;
45 static int futex_private_flag = FUTEX_PRIVATE_FLAG;

47 namespace {
48 static struct InitModule {
49  InitModule() {
50  int x = 0;
51  // futexes are ints, so we can use them only when
52  // that's the same size as the lockword_ in SpinLock.
53 #ifdef __arm__
54  // ARM linux doesn't support sys_futex1(void*, int, int, struct timespec*);
55  have_futex = 0;
56 #else
57  have_futex = (sizeof (Atomic32) == sizeof (int) &&
58  sys_futex(&x, FUTEX_WAKE, 1, 0) >= 0);
59 #endif
60  if (have_futex &&
61  sys_futex(&x, FUTEX_WAKE | futex_private_flag, 1, 0) < 0) {
62  futex_private_flag = 0;
63  }
64  }
65 } init_module;
66 
67 } // anonymous namespace

73 void SpinLockDelay(volatile Atomic32 *w, int32 value, int loop) {
74  if (loop != 0) {
75  int save_errno = errno;
76  struct timespec tm;
77  tm.tv_sec = 0;
78  if (have_futex) {
79  tm.tv_nsec = base::internal::SuggestedDelayNS(loop);
80  } else {
81  tm.tv_nsec = 2000001; // above 2ms so linux 2.4 doesn't spin
82  }
83  if (have_futex) {
84  tm.tv_nsec *= 16; // increase the delay; we expect explicit wakeups
85  sys_futex(reinterpret_cast<int *>(const_cast<Atomic32 *>(w)),
86  FUTEX_WAIT | futex_private_flag,
87  value, reinterpret_cast<struct kernel_timespec *>(&tm));
88  } else {
89  nanosleep(&tm, NULL);
90  }
91  errno = save_errno;
92  }
93 }
94 
95 void SpinLockWake(volatile Atomic32 *w, bool all) {
96  if (have_futex) {
97  sys_futex(reinterpret_cast<int *>(const_cast<Atomic32 *>(w)),
98  FUTEX_WAKE | futex_private_flag, all? INT_MAX : 1, 0);
99  }
100 }
